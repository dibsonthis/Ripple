var xPos = 0
var yPos = 10

[xPos, yPos]::onChange = (curr, old, name, info) => {
	println(name + " changed from " + string(old) + " to " + string(curr) + " on line " + string(info.line) + ", column " + string(info.column) + " in file '" + info.filename + "'")
}

for (1..15, i, v) {
	xPos = xPos + v
	yPos = yPos - v
}
